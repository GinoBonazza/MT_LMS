get_unique_top_markers <- function(markers, n = 10, cluster_col = "cluster", gene_col = "gene") {
  
  markers <- markers %>%
    dplyr::arrange(.data[[cluster_col]], dplyr::desc(avg_log2FC))
  
  clusters <- unique(markers[[cluster_col]])
  used_genes <- character(0)
  out <- list()
  
  for (cl in clusters) {
    
    cl_markers <- markers %>%
      dplyr::filter(.data[[cluster_col]] == cl) %>%
      dplyr::filter(!.data[[gene_col]] %in% used_genes) %>%
      dplyr::slice_head(n = n)
    
    used_genes <- c(used_genes, cl_markers[[gene_col]])
    out[[as.character(cl)]] <- cl_markers
  }
  
  dplyr::bind_rows(out)
}
